\begin{equation*} =\int\limits_{0}^{a}e^{-st}[f(t).u(t-a)]\,dt+\int\limits_{a}^{\infty}e^{-st}[f(t).u(t-a)]\,dt \end{equation*}
\begin{equation*} = 0+ \int\limits_{a}^{\infty}e^{-st}f(t).1\,dt \end{equation*}
\begin{equation*} = \int\limits_{0}^{\infty}e^{-s(y+a)}f(y+a)\,dy= e^{-as}\int\limits_{0}^{\infty}e^{-sy}f(y+a)\,dy = e^{-as}F(s+a) \end{equation*}
