Some Useful Formulae:.

  1. \begin{equation*} \int\limits_{0}^{2\pi} \sin nx \, dx = 0. \end{equation*}
  2. \begin{equation*} \int\limits_{0}^{2\pi} \cos nx \, dx = 0. \end{equation*}
  3. \begin{equation*} \int\limits_{0}^{2\pi} \sin^{2} nx\, dx = \pi. \end{equation*}
  4. \begin{equation*} \int\limits_{0}^{2\pi} \cos nx \,dx = \pi. \end{equation*}
  5. \begin{equation*} \int\limits_{0}^{2\pi} \sin nx .\sin mx\, dx = 0. \end{equation*}
  6. \begin{equation*} \int\limits_{0}^{2\pi} \cos nx .\cos mx\, dx = 0. \end{equation*}
  7. \begin{equation*} \int\limits_{0}^{2\pi} \sin nx .\cos mx\, dx = 0. \end{equation*}
  8. \begin{equation*} \int\limits_{0}^{2\pi} \sin nx .\cos nx\, dx = 0 \end{equation*}
  9. \begin{equation*} \sin n\pi = 0. \end{equation*}
  10. \begin{equation*} \cos n\pi = (-1)^{n}, \quad \text{where n is any integer}. \end{equation*}
  11. \begin{equation*} u v\, dx = u v_{1}-u' v_{2}+u''v_{3}-\cdots \end{equation*}
    where \(u' = \frac{\,du}{\,dx}, u'' = \frac{\,d^{2}u}{\,dx^{2}}, \cdots;\) \(v_{1}=\int v\, dx, v_{2}= \int v_{1}\,dx, \cdots\text{.}\)
  12. \begin{equation*} \int e^{ax} \cos bx\, dx = \frac{e^{ax}}{a^{2}+b^{2}}(a\cos bx+b\sin bx) \end{equation*}
  13. \begin{equation*} \int e^{ax} \sin bx\, dx = \frac{e^{ax}}{a^{2}+b^{2}}(a\sin bx-b\cos bx) \end{equation*}
  14. \begin{equation*} \int\limits_{-\pi}^{\pi} \sin nx .\sin mx\, dx \end{equation*}
    \begin{equation*} = \begin{cases} =0, \quad m=0 =\pi \delta_{mn}, \quad m\neq 0 \end{cases} \end{equation*}
  15. \begin{equation*} \int\limits_{-\pi}^{\pi} \sin nx .\cos mx\, dx = 0. \end{equation*}
  16. \begin{equation*} \int\limits_{-\pi}^{\pi} \cos nx .\cos mx\, dx = \begin{cases} =2\pi, \quad m==n=0\\ =\pi \delta_{mn}, \quad m\neq 0 \end{cases} \end{equation*}
  17. \begin{equation*} \sin A\sin B = \frac{1}{2}\left[\cos (A-B) - \cos(A+B)\right]; \end{equation*}
  18. \begin{equation*} \cos A\cos B = \frac{1}{2}\left[\cos (A-B) + \cos(A+B)\right]; \end{equation*}
  19. \begin{equation*} \sin A\cos B = \frac{1}{2}\left[\sin (A-B) + \sin (A+B)\right]. \end{equation*}
