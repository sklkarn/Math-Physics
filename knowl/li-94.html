from sympy.interactive import printing

from sympy import *
import sympy as sp
t,y =sp.symbols(’t,y’)
a=sp.Symbol(’a’,positive=True)
y=sp.Function(’y’)(t)
diffeq=Eq(y.diff(t,2)+a**2*y,0)
display(diffeq)
dsolve(diffeq,y)
import numpy as np
from scipy.integrate import odeint
import matplotlib.pyplot as plt
def f(u,t): return (u[1], -5*u[0])
yo=[0,4]
xs=np.arange(0,5,0.1)
us=odeint(f,yo,xs)
ys=us[:,0]
plt.plot(xs,ys, ’-’)
plt.plot(xs,ys, ’ro’)
plt.title(’(D**2+5)y=0’)
plt.xlabel(’t values’)
plt.ylabel(’y values’)
plt.show()
