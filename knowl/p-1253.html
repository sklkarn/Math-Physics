\begin{equation*} u(x,y) = \frac{4T}{\pi}\sum\limits_{r=1}^{\infty}\frac{1}{2^{r+1}} \sin\frac{(2r+1)\pi y}{a}e^{-\frac{(2r+1)\pi x}{a}} \end{equation*}
