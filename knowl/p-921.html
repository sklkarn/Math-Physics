  1. \begin{equation*} \int\limits_{0}^{\infty}\frac{\sin ax}{x}\,dx = \frac{\pi}{2}, \quad a \gt 0 \end{equation*}
  2. \begin{equation*} \int\limits_{0}^{\infty}e^{-ax} \sin bx \,dx = \frac{b}{a^{2}+b^{2}} \end{equation*}
  3. \begin{equation*} \int\limits_{0}^{\infty}e^{-ax} \cos bx \,dx = \frac{a}{a^{2}+b^{2}} \end{equation*}
  4. \begin{equation*} \int\limits_{0}^{\infty}e^{-x^{2}}\,dx = \frac{\sqrt{\pi}}{2} \end{equation*}
  5. \begin{equation*} \int\limits_{-\infty}^{\infty}e^{-x^{2}}\,dx = 1 \end{equation*}
  6. \begin{equation*} \Gamma_{n}=a^{n}\int\limits_{0}^{\infty}e^{-ax}x^{n-1}\,dx, \quad \text{the Gamma function.} \end{equation*}
