from scipy.special import j1
import numpy as np
import scipy as sp
import matplotlib.pyplot as plt
x = np.linspace(0, 20,100)
for n in range(4):
y = sp.special.jn(n,x)
plt.plot(x, y, label=r’$J_{}(x)$’.format(n))
plt.title("Bessel’s Function")
plt.xlabel("x")
plt.ylabel(r’$J_m(x)$’)
plt.legend(loc=’lower right’)
plt.show()
