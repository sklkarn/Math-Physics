import numpy as np
from scipy.special import eval_legendre
import matplotlib.pyplot as plt
from scipy.linalg import det
N = range(0, 6)
eval_legendre(N, 0)
X = np.linspace(-1, 1)
for n in range(0, 6):
y = eval_legendre(n, X)
plt.plot(X, y, label=r'$P_{}(x)$'.format(n))
plt.title("Legendre Polynomials")
plt.xlabel("x")
plt.ylabel(r'$P_n(x)$')
plt.legend(loc='lower right')
plt.show()
