\begin{equation*} \vec{v} = v_{r}\hat{e}_{r}+v_{\theta}\hat{e}_{\theta}+v_{\phi}\hat{e}_{\phi}; \end{equation*}
\begin{equation*} \vec{a}= a_{r}\hat{e}_{r}+a_{\theta}\hat{e}_{\theta}+a_{\phi}\hat{e}_{\phi}. \end{equation*}
where, \(v_{r} = \dot{r}, \hspace{4pt} v_{\theta} = r\dot{\theta}, \hspace{4pt} v_{\phi} = r\sin\theta\dot{\phi}\)
\begin{equation*} \text{and} \quad a_{r}=\ddot{r}-r\theta^{2}-r\sin^{2}\theta\dot{\phi}^{2}; \end{equation*}
\begin{equation*} a_{\theta}=\frac{1}{r}\frac{d}{dt}(r^{2}\dot{\theta})-r\sin\theta\cos\phi\dot{\phi}^{2}, \end{equation*}
\begin{equation*} a_{\phi}=\frac{1}{r\sin\theta}\frac{d}{dt}(r^{2}\sin^{2}\theta\dot{\phi}). \end{equation*}
