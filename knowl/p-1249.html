\begin{equation*} \frac{\partial u}{\partial t}= h^{2} \frac{\partial^{2}u}{\partial x^{2}}, 0 \lt x \lt \pi, t \lt 0. \end{equation*}
under the boundary conditions
\begin{equation*} \left.\frac{\partial u}{\partial x}\right\vert_{(0,t)}=0=\left.\frac{\partial u}{\partial x}\right\vert_{(\pi,t)}, \end{equation*}
and
\begin{equation*} \left.\frac{\partial u}{\partial x}\right\vert^{(x,0)}=\sin x. \end{equation*}
