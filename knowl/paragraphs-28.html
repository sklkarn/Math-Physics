\begin{equation*} \int\limits_{-\infty}^{\infty}e^{-x^{2}}H_{m}(x)H_{n}(x)\,dx =\sqrt{\pi}\left[2^{n-1}n!\delta_{m,n-1}+2^{n}(n+1)!\delta_{m,n+1}\right] \end{equation*}
