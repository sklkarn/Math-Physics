\begin{equation*} \vec{\alpha}\cdot (\vec{\beta}\times \vec{\gamma}) = \begin{bmatrix} 1 & -2 & 3 \\ -2 & 3 & -4 \\ 0 & -1 & 2 \end{bmatrix} \end{equation*}
\begin{equation*} = 1(6-4)+2(-4)+3(2) =2-8+6 = 0. \end{equation*}
