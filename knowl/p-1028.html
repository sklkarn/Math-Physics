\begin{equation*} \mathscr{L}\{f(t)\} = \int\limits_{0}^{\infty}e^{-st} f(t)\,dt \end{equation*}
\begin{equation*} = \int\limits_{0}^{T}e^{-st} f(t)\,dt+ \int\limits_{T}^{2T}e^{-st} f(t)\,dt+ \int\limits_{2T}^{3T}e^{-st} f(t)\,dt+\cdots \end{equation*}
\begin{equation*} =\sum\limits_{n=0}^{\infty}\int\limits_{nT}^{(n+1)T}e^{-st} f(t)\,dt \end{equation*}
