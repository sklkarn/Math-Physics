\begin{align*} x \amp = r cos\alpha\\ y \amp = r cos\beta\\ z \amp = r cos\gamma \\ Hence, \quad cos\alpha = \dfrac{x}{r} \amp = \dfrac{x}{\sqrt{x^{2}+y^{2}+z^{2}}} =l \\ or, \quad \cos\beta = \dfrac{y}{r} \amp = \dfrac{y}{\sqrt{x^{2}+y^{2}+z^{2}}} = m \\ or, \quad \cos\gamma = \dfrac{z}{r} \amp = \dfrac{z}{\sqrt{x^{2}+y^{2}+z^{2}}} = n. \end{align*}
